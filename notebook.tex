  \item
    (x,y) =\textgreater{} \{ ... \}
  \item
    x =\textgreater{} \{ ... \}
  \item
    () =\textgreater{} \{ ... \} // \_ =\textgreater{}
  \item
    x =\textgreater{} \{ return x * x; \}
  \item
    x =\textgreater{} x * x;
  \end{itemize}
\end{itemize}
